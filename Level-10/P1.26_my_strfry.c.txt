#include<stdio.h>
#include<string.h>
#define N 50  

char *my_strfry(char *str);

int main()
{
    char str[N],*ptr;
    printf("Enter string: ");
    fgets(str,N,stdin);
    ptr = my_strfry(str);
    printf("String after swapping characters randomly : %s\n",ptr);
   
    return 0;
}

char *my_strfry(char *str)
{
    int i=0,j=1,count=0,len;
    char ch;
    len = strlen(str)-1;
    for(i=0,j=1;count<(len/2); i=i+2,j=j+2)
    {
       ch = str[i];
       str[i] = str[j];
       str[j] = ch;
       count++;
    }
    return str;
}
