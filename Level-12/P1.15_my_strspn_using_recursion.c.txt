#include<stdio.h>
#define N 50 

int my_strspn(char *str1,char *str2,int i,int len);

int main()
{
    char str1[N],str2[N];
    printf("Enter string1: ");
    fgets(str1,N,stdin);
    printf("Enter string2: ");
    fgets(str2,N,stdin);
    int len = my_strspn(str1,str2,0,0);
    printf("Length of substring : %d\n",len);
    return 0;
}

int my_strspn(char *str1,char *str2,int i,int len)
{
    if(str1[i] != '\0')
    {
        int j = 0,flag = 0;
        if(str1[i] == str2[j])
        {
            while(str2[j] != '\0' && str2[j] != '\n')
            {
                if(str1[i+j] != str2[j])
                {
                    flag = 1;
                    break;
                }
                len++;
                j++;
            }
            if(flag == 1)
            {
                j = len = flag = 0;
            }
        } 
    }
    else
    return len;
    my_strspn(str1,str2,++i,len);
}
