#include <stdio.h>
#include <string.h>
#define  N 100 

int main()
{
    char str[N],str1[N];
    printf("Enter string1: ");
    fgets(str,N,stdin);
    printf("Enter string2: ");
    fgets(str1,N,stdin);
    int res = strcasecmp(str,str1);
    if(res == 0)
    printf("Strings are equal\n");
    else
    printf("Strings are not equal\n");

    return(0);
}
