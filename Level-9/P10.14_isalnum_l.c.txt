#include<stdio.h>
#include<ctype.h>
#include<locale.h>

int main()
{
    locale_t loc = newlocale(LC_ALL_MASK,"en_US.UTF-8",(locale_t)0);
    if(loc == (locale_t)0)
    {
        perror("newlocale");
        return 1;
    }
    
    char ch = 'B';
    if(isalnum_l(ch,loc))
    printf("%c is alphanumeric\n",ch);
    else
    printf("Not alphanumeric\n");
    freelocale(loc);
    return 0;
}
