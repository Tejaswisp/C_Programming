#include<stdio.h>
#include<stdarg.h>
#include <errno.h>
#include <fcntl.h>
#include <unistd.h>

void print(int fd,char *n, ...);

int main()
{
    int fd;
    fd = open("abc.txt",O_WRONLY | O_CREAT);
    print(fd,"args %d",10);
    
    return 0;
}

void print(int fd,char *n, ...)
{
    va_list ptr;
    va_start(ptr,n);
    vdprintf(fd,n,ptr);
    va_end(ptr);
}
