int my_strcasecmp(char *str1,char *str2,int i)
{
    if(str1[i] != '\0' || str2[i] != '\0')
    {
        if(str1[i]>=65 && str1[i]<=90)
        str1[i] = str1[i]+32;
        if(str2[i] >= 65 && str2[i]<=90)
        str2[i] = str2[i]+32;
        
        if(str1[i] != str2[i])
        return 1;
        else
        my_strcasecmp(str1,str2,++i);
    }
    else
    return 0;
}
