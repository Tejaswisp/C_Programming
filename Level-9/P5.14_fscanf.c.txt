#include<stdio.h>
#include<stdlib.h>
#define N 200

int main()
{
    FILE *fp;
    char buffer[N];
    fp = fopen("file1.txt","r");
    if(fp == NULL)
    {
        printf("Error in opening file\n");
        exit(1);
    }
    printf("File opened successfully\n");
    while(fscanf(fp,"%*s %s",buffer) == 1)
    printf("%s\n",buffer);
   
    fclose(fp);
    return 0;   
}
