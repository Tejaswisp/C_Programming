/* â€¢	ftell - return a file offset in a stream  */

#include<stdio.h>
#include<stdlib.h>
#define N 50

int main()
{
    FILE *fp;
    char buffer[N];
    fp = fopen("file1.txt","r");
    if(fp == NULL)
    {
        printf("Error in opening file\n");
        exit(1);
    }
    printf("File opened successfully\n");
    fscanf(fp,"%*s %s",buffer);
    printf("%ld\n",ftell(fp));
   
    fclose(fp);
    return 0;   
}
