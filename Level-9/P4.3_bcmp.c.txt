#include<stdio.h>
#include<string.h>

int main()
{
    float arr1[3] = {1.2,2.3,3};
    float arr2[3] = {1.2,2.4,3};
    int n;
    printf("Enter n value: ");
    scanf("%d",&n);
    int res = bcmp(arr1,arr2,n);
    if(res == 0)
    printf("Strings are equal\n");
    else
    printf("Strings are not equal\n");
    return 0;
}
