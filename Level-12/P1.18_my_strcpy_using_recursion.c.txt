/*â€¢	strcpy - copy a string and return a pointer to the end of the result,implement using recursion*/

#include<stdio.h>
#include<string.h>
#define N 50 

void my_strcpy(char *str1,char *str2,int i);

int main()
{
    char str1[N],str2[N];
    printf("Enter string : ");
    fgets(str2,N,stdin);
    my_strcpy(str1,str2,0);
    printf("%s\n",str1);
    return 0;
}

void my_strcpy(char *str1,char *str2,int i)
{
    if(str2[i] == '\0')
    return;
    else
    {
        str1[i] = str2[i];
        my_strcpy(str1,str2,++i);
    }
}
