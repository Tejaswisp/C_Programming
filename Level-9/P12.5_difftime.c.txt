#include<stdio.h>
#include<time.h>
#include<unistd.h>

int main()
{
    int i = 0,n=10;
    time_t t1,t2;
    time(&t1);
    
    while(i < n)
    {
        sleep(1);
        i++;
    }
    time(&t2);
    
    printf("%f seconds difference",difftime(t2,t1));
    return 0;
}
