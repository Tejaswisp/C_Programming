#include <stdio.h>
#include <string.h>
#define  N 100 

int main()
{
    char str[N],str1[N];
    printf("Enter string1: ");
    fgets(str,N,stdin);
    printf("Enter string2: ");
    fgets(str1,N,stdin); 
    char *s = str;
    char *p;
    while(p = strtok_r(s,str1,&s))
    {
        printf("%s\n",p);
    }
    
    return(0);
}
