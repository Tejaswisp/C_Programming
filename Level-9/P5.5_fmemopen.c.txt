/*â€¢	fmemopen - open a memory buffer stream  */

#include<stdio.h>
#include <errno.h>
#include <fcntl.h>
#include <unistd.h>
#define N 50 

int main()
{
    FILE *fp;
    char buffer[N];
    
    fp = fmemopen(buffer,N,"r");
    buffer[0] = 'A';
    printf("%s\n",buffer);
    
    fclose(fp);
    return 0;
}
