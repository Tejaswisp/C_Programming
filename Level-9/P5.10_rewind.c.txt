/* â€¢	rewind - reset the file position indicator in a stream  */

#include<stdio.h>
#include<stdlib.h>
#define N 50

int main()
{
    FILE *fp;
    char buffer[N];
    fp = fopen("file1.txt","r");
    if(fp == NULL)
    {
        printf("Error in opening file\n");
        exit(1);
    }
    printf("File opened successfully\n");
    fscanf(fp,"%*s %s",buffer);
    printf("%s\n",buffer);
    rewind(fp);
    fscanf(fp,"%*s %s",buffer);
    printf("%s\n",buffer);
   
    fclose(fp);
    return 0;   
}
