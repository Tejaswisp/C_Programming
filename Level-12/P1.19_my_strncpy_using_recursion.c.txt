#include<stdio.h>
#include<string.h>
#define N 50 

void my_strncpy(char *str1,char *str2,int i,int n);

int main()
{
    char str1[N],str2[N];
    int n;
    printf("Enter string : ");
    fgets(str2,N,stdin);
    printf("Enter n value: ");
    scanf("%d",&n);
    my_strncpy(str1,str2,0,n);
    printf("%s\n",str1);
    return 0;
}

void my_strncpy(char *str1,char *str2,int i,int n)
{
    if(i > n-1)
    {
        return;
        str1[i] = '\0';
    }
    else
    {
        str1[i] = str2[i];
        my_strncpy(str1,str2,++i,n);
    }
}
