#include<stdio.h>
#include<stdarg.h>

char buffer[50];

int print(char *n, ...);

int main()
{
    print("args %d %s",10,"HI");
    printf("%s",buffer);
    
    return 0;
}

int print(char *n, ...)
{
    va_list ptr;
    va_start(ptr,n);
    int res = vsprintf(buffer,n,ptr);
    va_end(ptr);
    return res;
}
