#include<stdio.h>
#define N 50 

int my_strspn(char *str1,char *str2);

int main()
{
    char str1[N],str2[N];
    printf("Enter string1: ");
    fgets(str1,N,stdin);
    printf("Enter string2: ");
    fgets(str2,N,stdin);
    int len = my_strspn(str1,str2);
    printf("Length of substring : %d\n",len);
    return 0;
}

int my_strspn(char *str1,char *str2)
{
    int len = 0,i,j=0,flag = 0;
    for(i=0; str1[i] != '\0'; i++)
    {
        if(str1[i] == str2[j])
        {
            while(str2[j] != '\0' && str2[j] != '\n')
            {
                if(str1[i+j] != str2[j])
                {
                    flag = 1;
                    break;
                }
                len++;
                j++;
            }
            if(flag == 1)
            {
                j = len = flag = 0;
            }
        }
    }
    return len;
}
