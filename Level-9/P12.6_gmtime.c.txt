/* â€¢	gmtime - convert a time value to a broken-down UTC time */

#include<stdio.h>
#include<time.h>
#define BST (+1)

int main()
{
    time_t t;
    struct tm *ptr;
    time(&t);
    ptr = gmtime(&t);
    printf("London : %2d:%02d\n", (ptr->tm_hour+BST)%24, ptr->tm_min);
    return 0;
}
