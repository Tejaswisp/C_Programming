#include<stdio.h>
#include<stdlib.h>
#define N 50

int main()
{
    FILE *fp;
    fpos_t position;
    char buffer[N];
    fp = fopen("file1.txt","w+");
    if(fp == NULL)
    {
        printf("Error in opening file\n");
        exit(1);
    }
    printf("File opened successfully\n");
    fgetpos(fp, &position);
    fputs("Hi", fp);
    fsetpos(fp,&position);
    fputs("Hello",fp);
   
    fclose(fp);
    return 0;   
}
