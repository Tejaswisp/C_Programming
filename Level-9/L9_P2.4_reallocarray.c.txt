#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n = 5;
    int *ptr = (int *)malloc(sizeof(int)*n);
    for(int i=0; i<n; i++)
    {
        ptr[i] = i;
    }
    ptr = reallocarray(ptr,2,sizeof(int));
    
    return(0);
}
