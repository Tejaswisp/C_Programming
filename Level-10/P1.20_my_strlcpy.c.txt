#include<stdio.h>
#define N 50 

void my_strlcpy(char *des,char *src,int n);

int main()
{
    char src[N],des[N];
    int n;
    printf("Enter source string: ");
    fgets(src,N,stdin);
    printf("Enter n value : ");
    scanf("%d",&n);
    my_strlcpy(des,src,n-1);
    printf("After copying : %s\n",des);
   
    return 0;
}

void my_strlcpy(char *des,char *src,int n)
{
    int i;
    for(i=0; i<n; i++)
    {
        des[i] = src[i];
    }
    des[i] = '\0';
}
